## 修改头像和昵称

```vue
<template>
	<view class="container">
		<view class="card">
			<view class="item flex align-center space-between">
				<view class="label">
					头像
				</view>
				<button class="avatar" open-type="chooseAvatar" @chooseavatar="chooseavatar">
					<image :src="userInfo.headimg" mode=""></image>
				</button>
			</view>
			<view class="item flex align-center space-between">
				<view class="label">
					昵称
				</view>
				<view class="value">
					<input type="nickname" v-model="userInfo.username" @change="getname" value="" placeholder="用户昵称" />
				</view>
			</view>
		</view>
		<view class="options flex center" @click="submit" >
			保存
		</view>
	</view>
</template>

<script>
	import {
		getInfo,update
	} from "@/api/user.js";
	import { upload } from "@/api/index.js";
	export default {
		data() {
			return {
				userInfo: {},
				sexRange: [{
					id: '男',
					name: '男'
				}, {
					id: '女',
					name: '女'
				}]
			}
		},
		onLoad() {
			this.getInfo();
		},
		methods: {
			getname(e){
				this.userInfo.username = e.detail.value;
			},
			chooseavatar(e){
				let that = this;
				console.log('event',e);
				upload({
					filePath: e.detail.avatarUrl,
				}).then(ras=>{
					that.userInfo.headimg = ras.info.url;
				})
				
			},
			submit(){
				update({
					data:{
						headimg:this.userInfo.headimg,
						username:this.userInfo.username,
					}
				}).then(res=>{
					if(res.code == 1){
						uni.showToast({
							title:'保存成功'
						});
						setTimeout(()=>{
							uni.navigateBack({
								
							})
						})
					}
				})
			},
			
			sexChange(e){
				this.userInfo.base_sex = this.sexRange[e.detail.value].name;
			},
			getInfo() {
				getInfo({
					data: {}
				}).then(res => {
					this.userInfo = res.data.user;
				})
			},
			goAddress() {
				uni.navigateTo({
					url: '/pages/me/address-list'
				})
			},
		}
	}
</script>
```
